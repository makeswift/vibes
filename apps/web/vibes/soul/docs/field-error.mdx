## Usage

{/* prettier-ignore-start */}

<CodeBlock lang="ts">{`
import { FieldError } from '@/vibes/soul/form/field-error';

function Usage() {
    return (
        <FieldError>You must accept the Terms & Conditions</FieldError>
    );
}
`}
</CodeBlock>
