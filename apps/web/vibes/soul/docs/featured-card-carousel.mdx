## Usage

{/* prettier-ignore-start */}

<CodeBlock lang="ts">{`
import { FeaturedCardCarousel } from '@/vibes/soul/sections/featured-card-carousel';

function Usage() {

    return (
      <FeaturedCardCarousel
          cards={cards}
          cta={{ label: 'Shop all', href: '/all' }}
          description="Lorem ipsum dolor sit amet, consectetur adipiscing elit, sed do eiusmod tempor incididunt ut labore et dolore magna aliqua."
          title="Categories"
      />
    );
}

const cards: CardContent[] = [
    {
      title: 'Mini Bar Bag',
      image: {
        src: 'https://rstr.in/monogram/vibes/mrlTNE1TJfB',
        alt: 'Mini Bar Bag',
      },
      href: '#',
    }
]

`}
</CodeBlock>

{/* prettier-ignore-end */}

## API Reference

### FeaturedCardCarouselProps

| Prop             | Type                                                                                             | Default |
| ---------------- | ------------------------------------------------------------------------------------------------ | ------- |
| `title*`         | `string`                                                                                         |         |
| `description`    | `string`                                                                                         |         |
| `cta`            | `Link` <Tooltip content="See Link below"/>                                                       |         |
| `cards*`         | [`Streamable`](docs/soul/streamable)`<CardContent[]>` <Tooltip content="See CardContent below"/> |         |
| `scrollbarLabel` | `string`                                                                                         |         |

### CardContent

| Prop     | Type                                         | Default |
| -------- | -------------------------------------------- | ------- |
| `title*` | `string`                                     |         |
| `image`  | `Image` <Tooltip content="See Image below"/> |         |
| `href`   | `string`                                     |         |

### Link

| Prop     | Type     | Default |

| `label*` | `string` |         |

### Image
