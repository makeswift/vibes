## Usage

{/* prettier-ignore-start */}

<CodeBlock lang="ts">{`
import { Breadcrumbs, type Breadcrumb } from '@/vibes/soul/sections/breadcrumbs';

function Usage() {

    const breadcrumbs: Breadcrumb[] = [
      {
        label: 'Home',
        href: '#1',
      },
      {
        label: 'Bags',
        href: '#2',
      },
      {
        label: 'Handle Bags',
        href: '#3',
      },
    ];

    return (
        <Breadcrumbs breadcrumbs={breadcrumbs} />
    );
}
`}
</CodeBlock>

{/* prettier-ignore-end */}
